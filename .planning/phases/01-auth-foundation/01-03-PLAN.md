  <files>rust/raptorflow-api/sqlx-data.json</files>
  <action>
Add PRL tables to sqlx-data.json:

1. `ripples` table:

   ```sql
   CREATE TABLE ripples (
       id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
       org_id UUID NOT NULL REFERENCES organizations(id),
       title TEXT NOT NULL,
       description TEXT,
       status TEXT NOT NULL DEFAULT 'active',  -- active, realized, decayed
       category TEXT,  -- idea, action, outcome
       created_by UUID REFERENCES users(id),
       created_at TIMESTAMPTZ DEFAULT NOW(),
       updated_at TIMESTAMPTZ DEFAULT NOW(),
       realized_at TIMESTAMPTZ,
       decayed_at TIMESTAMPTZ
   );
   ```

2. `ripple_edges` table:

   ```sql
   CREATE TABLE ripple_edges (
       id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
       org_id UUID NOT NULL REFERENCES organizations(id),
       source_ripple_id UUID NOT NULL REFERENCES ripples(id),
       target_ripple_id UUID NOT NULL REFERENCES ripples(id),
       relationship TEXT NOT NULL,  -- causes, supports, conflicts, relates
       created_at TIMESTAMPTZ DEFAULT NOW(),

       UNIQUE(source_ripple_id, target_ripple_id, relationship)
   );
   ```

3. `agent_essences` table:

   ```sql
   CREATE TABLE agent_essences (
       id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
       org_id UUID NOT NULL REFERENCES organizations(id),
       name TEXT NOT NULL,
       personality JSONB NOT NULL,
       memory JSONB DEFAULT '[]',
       created_at TIMESTAMPTZ DEFAULT NOW(),
       updated_at TIMESTAMPTZ DEFAULT NOW()
   );
   ```

4. Enable RLS on all three tables:

   ```sql
   ALTER TABLE ripples ENABLE ROW LEVEL SECURITY;
   ALTER TABLE ripple_edges ENABLE ROW LEVEL SECURITY;
   ALTER TABLE agent_essences ENABLE ROW LEVEL SECURITY;

   CREATE POLICY tenant_isolation ON ripples
       FOR ALL USING (org_id = current_setting('app.current_org_id', true)::uuid);
   CREATE POLICY tenant_isolation ON ripple_edges
       FOR ALL USING (org_id = current_setting('app.current_org_id', true)::uuid);
   CREATE POLICY tenant_isolation ON agent_essences
       FOR ALL USING (org_id = current_setting('app.current_org_id', true)::uuid);
   ```

5. Indexes:

   ```sql
   CREATE INDEX idx_ripples_org_status ON ripples(org_id, status);
   CREATE INDEX idx_ripples_created_by ON ripples(created_by);
   CREATE INDEX idx_edges_source ON ripple_edges(source_ripple_id);
   CREATE INDEX idx_edges_target ON ripple_edges(target_ripple_id);
   CREATE INDEX idx_essences_org ON agent_essences(org_id);
   ```

6. Add to sqlx-data.json

  <name>Task 4: Create decay policy engine</name>
  <files>rust/raptorflow-api/src/prl/decay.rs</files>
  <action>
Create decay policy engine:

1. Create `prl/decay.rs`:

   ```rust
   #[derive(Debug, Clone, Serialize, Deserialize)]
   pub struct DecayPolicy {
       pub name: String,
       pub condition: DecayCondition,
       pub action: DecayAction,
   }

   #[derive(Debug, Clone, Serialize, Deserialize)]
   #[serde(tag = "type")]
   pub enum DecayCondition {
       StaleDays { days: i32 },
       NoConnections { min_edges: i32 },
   }

   #[derive(Debug, Clone, Serialize, Deserialize)]
   #[serde(tag = "type")]
   pub enum DecayAction {
       MarkDecayed,
       Archive,
   }

   pub struct DecayEngine {
       policies: Vec<DecayPolicy>,
   }

   impl DecayEngine {
       pub fn new() -> Self {
           // Default policy: ripples with no connections for 30 days decay
           let policies = vec![
               DecayPolicy {
                   name: "stale_no_connections".into(),
                   condition: DecayCondition::NoConnections { min_edges: 1 },
                   action: DecayAction::MarkDecayed,
               },
               DecayPolicy {
                   name: "stale_never_updated".into(),
                   condition: DecayCondition::StaleDays { days: 90 },
                   action: DecayAction::MarkDecayed,
               },
           ];
           Self { policies }
       }

       /// Run decay policies against all active ripples for an org
       pub async fn run(&self, pool: &PgPool, org_id: Uuid) -> Result<DecayReport>

       /// Check single ripple against policies
       pub async fn check_ripple(&self, pool: &PgPool, ripple_id: Uuid) -> Result<Vec<PolicyResult>>
   }
   ```

2. DecayReport:

   ```rust
   pub struct DecayReport {
       pub ripples_processed: i32,
       pub decayed: i32,
       pub archived: i32,
       pub details: Vec<PolicyResult>,
   }

   pub struct PolicyResult {
       pub ripple_id: Uuid,
       pub policy_name: String,
       pub triggered: bool,
   }
   ```
